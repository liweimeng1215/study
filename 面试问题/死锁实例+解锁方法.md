# 死锁实例+解锁方法

```go
package main

import (
	"fmt"
	"sync"
)

var muA sync.Mutex
var muB sync.Mutex
var ch = make(chan int, 2)

func handlerB() {
	muB.Lock()
	fmt.Println("[B] handle ...")
	muB.Unlock()
	ch <- 2
}

func sendReqToB() {
	handlerB()
}

func handlerA() {
	muA.Lock()
	fmt.Println("[A] handle ...")
	muA.Unlock()
	ch <- 1
}

func sendReqToA() {
	handlerA()
}

func A() {
	muA.Lock()
	fmt.Println("[A] send request to B ...")
	//muA.Unlock() // 解锁方法
	sendReqToB()
	//muA.Lock() // 解锁方法
	fmt.Println("[A] get response")
	muA.Unlock()
}

func B() {
	muB.Lock()
	fmt.Println("[B] send request to A ...")
	//muB.Unlock() // 解锁方法
	sendReqToA()
	//muB.Lock() // 解锁方法
	fmt.Println("[B] get response")
	muB.Unlock()
}

func main() {
	go B()
	go A()
	for i := 0; i < 2; i++ {
		t := <-ch
		fmt.Printf("[main] get %v\n", t)
	}
}

```
